    const Companies = ({
    searchEmailTender,
    rowEventhandlers,
    companyId,
    emailReferenceId,
    getDetails,
    selectedTenderRowId,
    isCircular,
}) => {
    // const [companyIdForRemark, setCompanyId] = useState(false);
    const [valueRadioAddCompany, setValueRadioAddCompany] = useState(1);
    const [formCompanyId, setFormCompanyId] = useState("");
    const [value, setValue] = useState("");
    const [visibleQuickModal, setVisibleQuickModal] = useState(false);
    // added
    const [isCircularModalOpen, setIsCircularModalOpen] = useState(false);

    const handleAddCompanyModal = () => {
        setIsAddCompanyModelOpen(false);
        setFormCompanyId(null);
    };

    const handleCircularModal = () => {
        setIsCircularModalOpen(false);
        setHideInternalMessage(true);
    };

    const onFinish = (values) => {
        let newValues = {
        ...values,
        ReferenceId: emailReferenceId,
        };

        setIsAddCompanyModelOpen(true);
        // dispatch(ReferenceCompanyManualAdd(newValues));
        // added
        setFormCompanyId(values?.CompanyId);
    };

    const onFinishAddCompany = (values) => {
        let remarksData = {
        remarks:
            valueRadioAddCompany === 1
            ? "Facebook"
            : valueRadioAddCompany === 2
            ? "Whatsapp"
            : valueRadioAddCompany === 3
            ? "Twitter"
            : valueRadioAddCompany === 4
            ? // ? values?.companyAddRemarks
                "Other"
            : null,
        };

        // for manual Company Add
        const api1 = {
            ReferenceId: emailReferenceId,
            CompanyId: formCompanyId,
        };

        // for remarks
        const api2 = {
            tenderBasicId: getDetails.id,
            contactCompanyId: formCompanyId,
            ModeofCommunication: remarksData?.remarks,
            remarks: values?.remarks,
        };

        try {
            if (isCircular) {
                dispatch(actionForReferenceCircularPlanningAdd(api1, api2));
            } else {
                dispatch(ReferenceRemarksCompanyAdd(api1, api2));
            }
        } catch (err) {
            console.log("error", err);
        } finally {
            setIsAddCompanyModelOpen(false);
            addCompanyModelForm.resetFields();
        }
    };

      // helper functions
    const renderRadioWithRemarks = (value, label, remarksFieldName) => (
        <Radio key={value} value={value}>
        {label}
        {valueRadioAddCompany === value && (
            <Form.Item name={remarksFieldName} rules={[{ required: true }]}>
            <Input.TextArea rows={4} placeholder="Remarks" />
            </Form.Item>
        )}
        </Radio>
    );


    return(
        <>
            <Modal
                title=""
                className=""
                visible={isAddCompanyModelOpen}
                onCancel={handleAddCompanyModal}
                footer={null}>
                <Form form={addCompanyModelForm} onFinish={onFinishAddCompany}>
                    <Form.Item>
                        <Radio.Group
                            onChange={onChangeAddCompany}
                            value={valueRadioAddCompany}
                            name="addupdateremarks"
                        >
                        <Space direction="vertical">
                            {renderRadioWithRemarks(1, "Facebook", "remarks")}
                            {renderRadioWithRemarks(2, "Whatsapp", "remarks")}
                            {renderRadioWithRemarks(3, "Twitter", "remarks")}
                            {renderRadioWithRemarks(4, "Other", "remarks")}
                        </Space>
                        </Radio.Group>
                    </Form.Item>
                    <Form.Item>
                        <Button
                            className="company-add-save-btn"
                            type="primary"
                            htmlType="submit"
                            shape="round"
                        >
                            Submit
                        </Button>
                        <Button
                            type="dotted"
                            onClick={() => {
                                setIsAddCompanyModelOpen(false);
                            }}
                            bordered={false}
                            shape="round"
                        >
                            Close
                        </Button>
                    </Form.Item>
                </Form>
            </Modal>
        </>
    )
}
