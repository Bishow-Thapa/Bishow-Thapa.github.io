    import { useEffect, useState } from "react";

import { Layout, Empty, Flex, Spin, Typography, Grid } from "antd";
import { useParams, useNavigate } from "react-router-dom"; // <-- add this
import { useGetForexByIdQuery } from "@features/forex/services/forexApi";

import DealsList from "@features/forex/components/DealsList";
import DealsListDetails from "@features/forex/components/DealsListDetails";
import "@features/forex/styles/deals.scss";
import { DealTypeEnum } from "@shared/constants/DealType";

const { Content } = Layout;
const { useBreakpoint } = Grid;

const AllDeals = () => {
  const navigate = useNavigate();
  const { id } = useParams();
  const { lg } = useBreakpoint();
  const { isCollapsedDealChat } = useSelector((state) => state.collapse);

  const [selectedItem, setSelectedItem] = useState(null);
  const [checkedItems, setCheckedItems] = useState([]);
  const [isChatNotification, setIsChatNotification] = useState(null);

  // Fetch deal by selected item or param id
  const {
    data: forexData,
    isLoading,
    error: forexError,
    isFetching,
  } = useGetForexByIdQuery(
    { id: selectedItem?.id || id }, // <-- use param id when no selected
    {
      skip: !(selectedItem?.id || id),
      refetchOnMountOrArgChange: true,
    }
  );

  const handleSelectDeal = (deal) => {
    setSelectedItem(deal);
    // update URL with deal ID
    navigate(`/deals/forex/${deal.id}`, { replace: false });
  };

  // When id is present (direct route), set it as selected
  // useEffect(() => {
  //   if (id) {
  //     setSelectedItem({ id });
  //   }
  // }, [id]);

  return (
    <Content className="deals-inbox">
      <div
        className={`list ${
          !lg && !isCollapsedDealChat
            ? "deal-listing--collapsed"
            : "deal-listing--expanded"
        }`}
      >
        <DealsList
          updatedDealData={forexData}
          selectedItem={selectedItem}
          setSelectedItem={setSelectedItem}
          // setSelectedItem={handleSelectDeal}
          onSelectDeal={handleSelectDeal}
          checkedItems={checkedItems}
          setCheckedItems={setCheckedItems}
          setIsChatNotification={setIsChatNotification}
        />
      </div>

      <div
        className={`listdetails ${
          !isCollapsedDealChat
            ? "deal-listing-details--expand"
            : "deal-listing-details--collapse"
        }`}
        style={{ overflow: "hidden" }}
      >
        {selectedItem && !forexError ? (
          isFetching ? (
            <Flex style={{ height: "100%" }} justify="center" align="center">
              <Spin size="large" />
            </Flex>
          ) : (
            <DealsListDetails
              dataFrom={DealTypeEnum.FOREX}
              dealData={forexData}
              isChatNotification={isChatNotification}
            />
          )
        ) : (
          <Flex style={{ height: "100%" }} justify="center" align="center">
            <Empty description={false} />
          </Flex>
        )}
      </div>
    </Content>
  );
};

export default AllDeals;

      path: APP_ROUTES.REPORTING_FCY_POSITION,
      element: <FcyPosition />,
      requiredpermissions: [PERMISSIONS.CAN_VIEW_TREASURY_REPORT],
    },
    {
      path: APP_ROUTES.TUTORIAL,
      element: <Tutorial />,
      // requiredpermissions: [PERMISSIONS.CAN_VIEW_TUTORIAL],
    },
    {
      path: APP_ROUTES.DEAL_TRANSACTION_TYPE,
      element: <DealTransactionType />,
      requiredpermissions: [PERMISSIONS.CAN_VIEW_DEAL_TRANSACTION_TYPE],
    },
    {
      path: APP_ROUTES.TUTORIAL_CATEGORY,
      element: <TutorialCategory />,
      requiredpermissions: [PERMISSIONS.CAN_VIEW_TUTORIAL],
    },
    {
      path: APP_ROUTES.TUTORIAL_SETTINGS,
      element: <TutorialSettings />,
      requiredpermissions: [PERMISSIONS.CAN_VIEW_TUTORIAL],
    },
    {
      path: APP_ROUTES.NOTIFICATION,
      element: <AllNotification />,
    },

    { path: "*", element: <NotFound /> },
  ];

  if (CONFIG.BUILD_MODE === DOMAIN_LISTS.INFOCOM) {
    // protectedRoutes.push({
    //   path: APP_ROUTES.FOREX_FORM_TEST,
    //   element: <ForexFormTest />,
    // });
    protectedRoutes.push(
      ...[
        {
          path: APP_ROUTES.FOREX_FORM_TEST,
          element: <ForexFormTest />,
        },
        {
          path: APP_ROUTES.TEST_BRANCH_CURRENCY,
          element: <BranchCurrencyRequest />,
        },
        {
          path: APP_ROUTES.TEST_BRANCH_MFT,
          element: <BranchMFTRequest />,
        },
      ]
    );
  }

  return (
    <Router>
      <Routes>
        <Route
          path="/"
          element={
            <RedirectRoute>
              <Login />
            </RedirectRoute>
          }
        ></Route>
        <Route path="/reset-password/:token" element={<ResetPassword />} />
        {protectedRoutes.map(({ path, element, requiredpermissions }) => (
          <Route
            key={path}
            path={path}
            element={
              <PrivateRoute requiredpermissions={requiredpermissions}>
                <MainLayout>{element}</MainLayout>
              </PrivateRoute>
            }
          />
        ))}
        <Route path="*" element={<NotFound />} />
      </Routes>
    </Router>
  );
};

export default AppRoutes;
