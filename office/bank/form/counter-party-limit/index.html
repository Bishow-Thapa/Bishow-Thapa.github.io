    useCounterPartyLimit.js

    import { useEffect, useMemo } from "react";

import { useLazyGetMMDExchangeRateQuery } from "@features/mmd/services/mmdformApi";
import { useGetCounterPartyTransactionLimitsQuery } from "@features/settings/services/settingApi";
import { MoneyMarketDealModeEnum } from "@features/mmd/utils/constants";
import logger from "@shared/utils/logger";

const useCounterPartyLimit = ({
  baseCurrencyId,
  counterpartyId,
  dealType,
  dealAmount,
}) => {
  const [
    getMMDExchangeRate,
    { data: MMDExchangeRateData, isLoading: isLoadingMMDExchageRate },
  ] = useLazyGetMMDExchangeRateQuery();

  const {
    data: counterPartyTransactionLimitData,
    isLoading: isCounterPartyTransactionLimitDataLoading,
    error: counterPartyTransactionLimitDataError,
  } = useGetCounterPartyTransactionLimitsQuery(
    {
      counterpartyId: counterpartyId,
    },
    {
      skip: !counterpartyId,
      refetchOnMountOrArgChange: true,
    }
  );

  const covertCurrencyToBase = useMemo(() => {
    const baseAmountRate = MMDExchangeRateData?.midRate;
    // Map
    const bankLimitData = counterPartyTransactionLimitData[0];

    if (baseCurrencyId !== bankLimitData?.currencyId) {
      return dealAmount * baseAmountRate;
    }
    return null;
  }, [
    MMDExchangeRateData,
    counterPartyTransactionLimitData,
    baseCurrencyId,
    dealAmount,
  ]);

  logger.info({
    msg: "HOOKS INSIDE LOGS ",
    covertCurrencyToBase,
    AMOUNT: dealAmount,
    MIDRATE: MMDExchangeRateData?.midRate,
  });

  // CurrencyExchangeRate
  useEffect(() => {
    if (
      baseCurrencyId &&
      counterpartyId &&
      dealType === MoneyMarketDealModeEnum.Lending
    ) {
      getMMDExchangeRate({
        currencyId: baseCurrencyId,
        counterpartyId: counterpartyId,
      });
    }
  }, [baseCurrencyId, counterpartyId, dealType, getMMDExchangeRate]);

  return {
    baseCurrencyData: MMDExchangeRateData,
    isBaseCurrencyDataLoading: isLoadingMMDExchageRate,
    limitData: counterPartyTransactionLimitData,
    isLimitDataLoading: isCounterPartyTransactionLimitDataLoading,
  };
};

export default useCounterPartyLimit;

    🧮 What Is "Rate Conversion"?

    Let’s say you receive an exchange rate like 1 USD = 2300 MMK, and you want to convert an amount (e.g., 100 USD) to MMK inside your hook.

    You can add the conversion logic right after fetching MMDExchangeRateData, and expose a helper function or pre-converted value from the hook.

    import { useEffect, useMemo } from "react";
import { useLazyGetMMDExchangeRateQuery } from "@features/mmd/services/mmdformApi";
import { useGetCounterPartyTransactionLimitsQuery } from "@features/settings/services/settingApi";
import { MoneyMarketDealModeEnum } from "@features/mmd/utils/constants";

const useCounterPartyLimit = ({
  baseCurrencyId,
  counterpartyId,
  dealType,
  amount = 0,
}) => {
  const [
    getMMDExchangeRate,
    { data: MMDExchangeRateData, isLoading: isLoadingMMDExchageRate },
  ] = useLazyGetMMDExchangeRateQuery();

  const {
    data: counterPartyTransactionLimitData,
    isLoading: isCounterPartyTransactionLimitDataLoading,
    error: counterPartyTransactionLimitDataError,
  } = useGetCounterPartyTransactionLimitsQuery(
    { counterpartyId },
    {
      skip: !counterpartyId,
      refetchOnMountOrArgChange: true,
    }
  );

  useEffect(() => {
    if (
      baseCurrencyId &&
      counterpartyId &&
      dealType === MoneyMarketDealModeEnum.Lending
    ) {
      getMMDExchangeRate({ currencyId: baseCurrencyId, counterpartyId });
    }
  }, [baseCurrencyId, counterpartyId, dealType, getMMDExchangeRate]);

  // 💱 Currency Conversion Logic (e.g. amount * rate)
  const convertedAmount = useMemo(() => {
    const rate = MMDExchangeRateData?.exchangeRate; // adjust key if different
    if (rate && amount > 0) {
      return amount * rate;
    }
    return null;
  }, [MMDExchangeRateData, amount]);

  return {
    baseCurrencyData: MMDExchangeRateData,
    isBaseCurrencyDataLoading: isLoadingMMDExchageRate,
    limitData: counterPartyTransactionLimitData,
    isLimitDataLoading: isCounterPartyTransactionLimitDataLoading,
    limitDataError: counterPartyTransactionLimitDataError,
    convertedAmount, // 👈 expose this to component
  };
};

export default useCounterPartyLimit;

    Usage

    const MyComponent = () => {
  const [amount, setAmount] = useState(100); // user input
  const { convertedAmount } = useCounterPartyLimit({
    baseCurrencyId: "USD",
    counterpartyId: "123",
    dealType: MoneyMarketDealModeEnum.Lending,
    amount,
  });

  return (
    <div>
      <Input
        type="number"
        value={amount}
        onChange={(e) => setAmount(Number(e.target.value))}
      />
      <p>
        Converted Amount: {convertedAmount ? convertedAmount.toFixed(2) : "N/A"}
      </p>
    </div>
  );
};
