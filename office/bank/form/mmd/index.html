    How to Separate Logic & UI in React Using Custom Hook

    Creating React custom hooks with Pieces

    You can create a custom hook in a separate file and use it in your components for better code organization and reusability. Here’s how you can do it with your useDealFormWatches logic:

    1. Create a new file for your hook (e.g., useDealFormWatches.js):

    // useDealFormWatches.js

import { Form } from "your-form-library"; // replace with your actual import path
import { DealFormEnum } from "your-enum-file"; // replace with your actual import path

export function useDealFormWatches(form) {
  return {
    dealCurrencyId: Form.useWatch(DealFormEnum?.DEAL_CURRENCY, form),
    freshOrRoll: Form.useWatch(DealFormEnum?.FRESH_ROLL, form),
    interestAmount: Form.useWatch(DealFormEnum?.INTEREST_AMOUNT, form),
    dealAmount: Form.useWatch(DealFormEnum?.DEAL_AMOUNT, form) ?? 0,
    valueDate: Form.useWatch(DealFormEnum?.VALUE_DATE, form) ?? 0,
    numberOfDays: Form.useWatch(DealFormEnum?.NUMBER_OF_DAYS, form) ?? 0,
    interestRate: Form.useWatch(DealFormEnum?.INTEREST_RATE, form) ?? 0,
    maturityDate: Form.useWatch(DealFormEnum?.MATURITY_DATE, form) ?? 0,
    basisOfCounting: Form.useWatch(DealFormEnum?.COUNTING_TYPE, form) ?? 0,
    totalAccumulateAmount: Form.useWatch(DealFormEnum?.TOTAL_VALUE, form) ?? 0,
  };
}

    import { Form } from "antd";
import { DealFormEnum } from "@features/mmd/utils/constants";

export default function useDealFormWatches(form) {
  // List of all fields you want to watch collectively
  const allFields = [
    DealFormEnum.DEAL_CURRENCY,
    DealFormEnum.FRESH_ROLL,
    DealFormEnum.INTEREST_AMOUNT,
    DealFormEnum.DEAL_AMOUNT,
    DealFormEnum.VALUE_DATE,
    DealFormEnum.NUMBER_OF_DAYS,
    DealFormEnum.INTEREST_RATE,
    DealFormEnum.MATURITY_DATE,
    DealFormEnum.COUNTING_TYPE,
    DealFormEnum.TOTAL_VALUE,
  ];

  return {
    // Watch all fields as an object (keys are field names, values are current values)
    allFields: Form.useWatch(allFields, form),

    // Individual watches for convenience
    dealCurrencyId: Form.useWatch(DealFormEnum.DEAL_CURRENCY, form),
    freshOrRoll: Form.useWatch(DealFormEnum.FRESH_ROLL, form),
    interestAmount: Form.useWatch(DealFormEnum.INTEREST_AMOUNT, form),
    dealAmount: Form.useWatch(DealFormEnum.DEAL_AMOUNT, form) ?? 0,
    valueDate: Form.useWatch(DealFormEnum.VALUE_DATE, form) ?? 0,

    basisOfCounting: Form.useWatch(DealFormEnum.COUNTING_TYPE, form) ?? 0,
    totalAccumulateAmount: Form.useWatch(DealFormEnum.TOTAL_VALUE, form) ?? 0,
  };
}

    How to improve and DRY (Don't Repeat Yourself) your code

    You can extract the common logic into a reusable function and call it from a single useEffect for each form. This improves readability and maintainability.

    Step 1: Extract common logic into a function

    function updateFormWithApprovals({
  form,
  approvalOptions,
  verifierOptions,
  setSelectedVerifier,
  setSelectedApprovals,
}) {
  const verifiersIdOnly = verifierOptions?.map((item) => item?.aspNetUserId);
  const verifiersList = findUsersByIds(
    verifiersIdOnly,
    verifierOptions,
    VERIFIER_TYPES.VERIFIERS
  );
  setSelectedVerifier(verifiersList);

  if (!isEmpty(verifiersIdOnly)) {
    form.setFieldsValue({
      [DealFormEnum.DEAL_APPROVERS]: verifiersIdOnly,
    });
  } else {
    form.setFieldsValue({
      [DealFormEnum.DEAL_APPROVERS]: null,
    });
  }

  const approversIdOnly = approvalOptions.map((item) => item?.aspNetUserId);
  const approversList = findUsersByIds(
    approversIdOnly,
    approvalOptions,
    VERIFIER_TYPES.APPROVERS
  );
  const firstApprover = approversList[0];
  setSelectedApprovals([firstApprover]);

  form.setFieldsValue({
    [DealFormEnum.FRESH_ROLL]: MoneyMarketDealTypeEnum.FRESH,
  });

  if (!isEmpty(approversIdOnly)) {
    form.setFieldsValue({
      [DealFormEnum.DEAL_VERIFIERS]:
        approversIdOnly[0] === verifiersIdOnly[0]
          ? approversIdOnly.length > 1
            ? approversIdOnly[1]
            : null
          : approversIdOnly[0],
    });
  } else {
    form.setFieldsValue({
      [DealFormEnum.DEAL_VERIFIERS]: null,
    });
  }
}

    Benefits of this approach

    • Single source of truth: The logic to update forms is in one place.

    • Less code duplication: Easier to maintain and less error-prone.

    • Clear separation: useEffect hooks just call the updater function with appropriate params.

    Additional tips

    • You might want to memoize updateFormWithApprovals with useCallback if you pass it around or if you want to prevent unnecessary re-renders.

    • Ensure that setSelectedVerifier and setSelectedApprovals are stable (e.g., from useState or useReducer) or memoized.
