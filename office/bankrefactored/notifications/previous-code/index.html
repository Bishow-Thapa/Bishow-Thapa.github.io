    import React, { useEffect, useState } from "react";
import { useSelector, useDispatch } from "react-redux";

  Avatar,

  Badge,

  Tabs,
} from "antd";
import { UserOutlined } from "@ant-design/icons";
import InfiniteScroll from "react-infinite-scroll-component";
import dayjs from "dayjs";
import { useNavigate } from "react-router-dom";
import { isEmpty } from "lodash";
const { Title, Text } = Typography;
// features
import {
  useGetSignalRsUnreadNotificationQuery,
  usePostSignalRsMarkasreadMutation,
} from "@features/notifications/services/chatApi";
import {
  resetBellNotifications,
  resetBellNotificationsCount,
} from "@features/notifications/services/chatSlice";
// shared
import CustomButton from "@shared/components/Button/CustomButton";
import { CONFIG } from "@shared/utils/config";
import { capitalizeFirstLetter } from "@shared/utils/capitalizeFirstLetter";
import { DealTypeBackendEnum } from "@shared/constants/DealType";
import { APP_ROUTES } from "@shared/routes";
import logger from "@shared/utils/logger";

const AllNotification = () => {
  const dispatch = useDispatch();
  const [loading, setLoading] = useState(false);
  const [checkedId, setCheckedId] = useState([]);
  const [data, setData] = useState([]);
  const appUserId = useSelector((state) => state.auth.appUserId);
  const [pageDetail, setPageDetail] = useState({
    pageNum: 1,
    pageMaxCount: 20,
    isFilteredUnread: false,
  });
  const [hasMore, setHasMore] = useState(true);
  const navigate = useNavigate();

  const {
    data: unreadNotifications,
    isLoading: loadingUnreadNotifications,
    isError: errorLoadingUnreadNotifications,
    isFetching: isFetchingNotification,
  } = useGetSignalRsUnreadNotificationQuery(
    {
      userId: appUserId,
      pagenum: pageDetail?.pageNum,
      pagesize: pageDetail?.pageMaxCount,
      filterUnRead: pageDetail?.isFilteredUnread,
    },
    {
      refetchOnMountOrArgChange: true,
    }
  );

  const [
    postSignalRsMarkasread,
    {
      isLoading: loadingPostSignalRsMarkasread,
      isSuccess: loadingPostSignalRsMarkasreadSuccess,
      isError: loadingPostSignalRsMarkasreadError,
      error: loadingPostSignalRsMarkasreadErrorMessage,
    },
  ] = usePostSignalRsMarkasreadMutation();

  const fetchData = (pagenum, pagesize, filterUnRead) => {
    setPageDetail((prev) => ({
      ...prev,
      pageNum: prev.pageNum + 1,
    }));
  };

  useEffect(() => {
    if (
      unreadNotifications &&
      unreadNotifications?.notifications?.length < pageDetail.pageMaxCount
    ) {
      setHasMore(false);
    }
  }, [unreadNotifications, pageDetail.pageMaxCount]);

  useEffect(() => {
    setData([]);
  }, []);

  useEffect(() => {
    if (unreadNotifications) {
      setData((prevData) => [
        ...prevData,
        ...unreadNotifications.notifications,
      ]);
    }
  }, [unreadNotifications]);

  // useEffect(() => {
  //   if (unreadNotifications) {
  //     setData(unreadNotifications);
  //   }
  // }, [unreadNotifications, pageDetail?.isFilteredUnread]);

  const onMarkAsReadSubmmit = async ({ readItem }) => {
    logger.info({ readItem: readItem });
    try {
      const dataToSubmit = {
        userId: appUserId,
        notificationId:
          readItem !== undefined
            ? [readItem]
            : !isEmpty(checkedId)
            ? checkedId
            : [0],
      };

      await postSignalRsMarkasread(dataToSubmit).unwrap();
      setData([]);
      dispatch(resetBellNotifications());
      dispatch(resetBellNotificationsCount());
    } catch (error) {
      console.error("Error marking notifications as read:", error);
    }
  };

  const handleOk = (item) => {
    if (!pageDetail?.isFilteredUnread) {
      logger.info({ setCheckedId: checkedId });
      onMarkAsReadSubmmit({ readItem: item?.id });
    }

    if (item?.dealType === DealTypeBackendEnum.FMD) {
      navigate(APP_ROUTES?.FOREX_DEAL, {
        state: {
          notificationId: item?.dealId,
          dealType: item?.dealType,
          formCreationId: item?.dealId,
          isChatNotification: item?.type === 1,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.MMD) {
      navigate(APP_ROUTES?.MMD_DEAL, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          formCreationId: item?.dealId,
          isChatNotification: item?.type === 1,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.MFD) {
      navigate(APP_ROUTES?.MFT_DEAL, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          formCreationId: item?.dealId,
          isChatNotification: item?.type === 1,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.NDF) {
      navigate(APP_ROUTES?.NDF_DEAL, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          formCreationId: item?.dealId,
          isChatNotification: item?.type === 1,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.FCD) {
      navigate(APP_ROUTES?.FORWARD_CONTRACT_DEAL, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          formCreationId: item?.dealId,
          isChatNotification: item?.type === 1,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.CURD) {
      navigate(APP_ROUTES?.BRANCH_REQUEST_CURRENCY, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          requestId: item?.requestId,
          subtype: item?.subtype,
          role: item?.role,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.MFTRD) {
      navigate(APP_ROUTES?.BRANCH_REQUEST_MFT, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          requestId: item?.requestId,
          subtype: item?.subtype,
          role: item?.role,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.CURI) {
      navigate(APP_ROUTES?.BRANCH_REQUEST_CURRENCY, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          requestId: item?.requestId,
          subtype: item?.subtype,
          role: item?.role,
        },
      });
      return;
    }
    if (item?.dealType === DealTypeBackendEnum.MFTRI) {
      navigate(APP_ROUTES?.BRANCH_REQUEST_MFT, {
        state: {
          notificationId: item?.id,
          dealType: item?.dealType,
          requestId: item?.requestId,
          subtype: item?.subtype,
          role: item?.role,
        },
      });
      return;
    }

    // navigate(item?.url, {
    //   state: {
    //     notificationId: item?.id,
    //     dealType: item?.dealType,
    //     dealId: item?.dealId,
    //     isChatNotification: item?.type === 1,
    //   },
    // });
  };

  const items = [
    {
      label: "Unread Notifications",
      key: "1",
    },
    {
      label: "All Notifications",
      key: "2",
    },
  ];

  const onChange = (checkedValues) => {
    setCheckedId(checkedValues);
  };

  return (
    <>
      <Flex justify="space-between" align="center">
        <Tabs
          defaultActiveKey="1"
          size={"large"}
          style={{ marginBottom: 5 }}
          items={items}
          onTabClick={(key) => {
            if (key === "1") {
              setPageDetail((prev) => ({
                ...prev,
                pageNum: 1,
                isFilteredUnread: false,
              }));
              setData([]);
              setHasMore(true);
            } else {
              setPageDetail((prev) => ({
                ...prev,
                pageNum: 1,
                isFilteredUnread: true,
              }));
              setData([]);
              setHasMore(true);
            }
          }}
        />
        {data?.length > 0 && !pageDetail?.isFilteredUnread && (
          <CustomButton
            type="primary"
            title={"Mark as Read"}
            onClick={onMarkAsReadSubmmit}
            loading={loadingPostSignalRsMarkasread}
          />
        )}
      </Flex>

      {/* <Flex justify="space-between" align="center" style={{ marginBottom: 16 }}>
        <Title level={4} style={{ margin: 0 }}>
          Unread notifications
        </Title>

        <CustomButton
          type="primary"
          title="Mark as Read"
          onClick={onMarkAsReadSubmmit}
          loading={loadingPostSignalRsMarkasread}
        />
      </Flex> */}

      <div
        id="scrollableDiv"
        style={{
          // height: "100%",
          height: "calc(100vh - 180px)",
          overflow: "auto",
          padding: "0 16px",
          // border: "1px solid rgba(140, 140, 140, 0.35)",
          backgroundColor: "#fff",
        }}
      >
        <InfiniteScroll
          dataLength={data?.length}
          next={fetchData}
          // hasMore={hasMore}
          hasMore={!isFetchingNotification && hasMore}
          loader={<Skeleton avatar paragraph={{ rows: 1 }} active />}
          endMessage={data?.length > 0 && <Divider plain>End</Divider>}
          scrollableTarget="scrollableDiv"
        >
          {!pageDetail?.isFilteredUnread && (
            <Checkbox.Group
              className="notification-checkbox-group"
              onChange={onChange}
            >
              <List
                className="notification-list"
                dataSource={data}
                renderItem={(item) => (
                  <List.Item key={item?.id}>
                    <Flex className="notification-checkbox"></Flex>
                    <Checkbox
                      value={item?.id}
                      style={{
                        marginRight: "8px",
                      }}
                    />
                    <List.Item.Meta
                      // avatar={<Avatar src={item?.picture?.large} />}
                      avatar={
                        item?.senderProfile != null ? (
                          <Avatar
                            size="small"
                            src={`${CONFIG.BASE_API_URL}${item?.senderProfile}`}
                          />
                        ) : (
                          <Avatar size="small" icon={<UserOutlined />} />
                        )
                      }
                      title={
                        <>
                          {!item?.isRead ? (
                            <Badge dot>
                              {capitalizeFirstLetter(item?.username)}
                            </Badge>
                          ) : (
                            capitalizeFirstLetter(item?.username)

                          <span
                            style={{
                              color: "#888",

                            {dayjs(item?.postedDateTime).format(
                              "YYYY-MM-DD h:mm:ss A"
                            )}

                        </>
                      }
                      description={item?.message}
                      onClick={() => {
                        handleOk(item);
                      }}
                    />
                  </List.Item>
                )}
              />
            </Checkbox.Group>
          )}
          {pageDetail?.isFilteredUnread && (
            <List
              className="notification-list"
              dataSource={data}
              renderItem={(item) => (
                <List.Item
                  key={item?.id}
                  onClick={() => {
                    handleOk(item);
                  }}
                >
                  <List.Item.Meta
                    // avatar={<Avatar src={item?.picture?.large} />}
                    avatar={
                      item?.senderProfile != null ? (
                        <Avatar
                          size="small"
                          src={`${CONFIG.BASE_API_URL}${item?.senderProfile}`}
                        />
                      ) : (
                        <Avatar size="small" icon={<UserOutlined />} />
                      )
                    }
                    title={
                      <>
                        {!item?.isRead ? (
                          <Badge dot>
                            {capitalizeFirstLetter(item?.username)}
                          </Badge>
                        ) : (
                          capitalizeFirstLetter(item?.username)
                        )}

                        <span
                          style={{
                            color: "#888",
                            fontSize: "12px",

                            "YYYY-MM-DD h:mm:ss A"
                          )}
                        </span>
                      </>
                    }
                    description={item?.message}
                  />
                </List.Item>
              )}
            />
          )}
        </InfiniteScroll>
      </div>
    </>
  );
};
export default AllNotification;
