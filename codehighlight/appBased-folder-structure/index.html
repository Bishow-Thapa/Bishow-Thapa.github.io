    APP BASED

    /app
  /layout.tsx             <-- Global layout for the whole app
  /page.tsx               <-- Home page (or root page)

  /auth
    /layout.tsx           <-- Layout for authentication-related pages (e.g., login, register)
    /login
      /page.tsx           <-- Login page
    /register
      /page.tsx           <-- Register page
    /forgot-password
      /page.tsx           <-- Forgot password page
    /reset-password
      /page.tsx           <-- Reset password page

  /blog
    /layout.tsx           <-- Layout for blog pages
    /page.tsx             <-- Blog homepage
    /[slug]
      /page.tsx           <-- Dynamic blog post page

  /dashboard
    /layout.tsx           <-- Layout for dashboard pages (e.g., profile, settings)
    /page.tsx             <-- Dashboard homepage
    /profile
      /page.tsx           <-- Profile page
    /settings
      /page.tsx           <-- Settings page

  /products
    /layout.tsx           <-- Layout for product-related pages
    /page.tsx             <-- Products homepage
    /[id]
      /page.tsx           <-- Dynamic product detail page

  /users
    /layout.tsx           <-- Layout for user-related pages
    /page.tsx             <-- Users homepage (user listing)
    /[id]
      /page.tsx           <-- Dynamic user profile page

  /404.tsx                <-- Custom 404 page
  /loading.tsx            <-- Global loading state for suspense
  /error.tsx              <-- Global error handling page
