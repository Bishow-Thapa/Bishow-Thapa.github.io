export const actionForCircularPlanningLog = (body) => {
    console.log("Fetching log list..."); // Debugging log
    return async (dispatch) => {
        dispatch(init(REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG));
        const response = await referencesService.getCircularPlanningLog(body);

        if (response.isSuccess) {
            dispatch(
                success(REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG, response.data)
            );
        } else if (!response.isSuccess) {
            dispatch(error(response.errorMessage));
        }
        dispatch(finish(REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG));
    };
};

    switch (action.type) {
        // log
        case REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG_INIT:
            return {
                ...state,
                getCircularPlanningResp: [],
                getCircularPlanningLoading: true,
            };

        case REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG_SUCCESS:
            return {
                ...state,
                getCircularPlanningResp: action.payload,
                getCircularPlanningLoading: false,
            };

        case REFERENCE__TYPES.GET_CIRCULAR_PLANNING_LOG_FINISH:
            return {
                ...state,
                getCircularPlanningLoading: false,
            };
    }
}

    import React, { useEffect, useReducer } from "react";
import { Table } from "antd";
import moment from "moment";

const LogList = (props) => {
    let { loading, data } = props;

    const columns = [
        {
            title: "S.N",
            dataIndex: "key",
            render: (_, __, index) => index + 1,
        },
        {
            title: "Remarks",
            dataIndex: "remarks",
            key: "remarks",
        },
        {
            title: "Creation Time",
            dataIndex: "creationTime",
            key: "creationTime",
            render: (text) => moment(text).format("YYYY-MM-DD"),
        },
    ];

    //   const dataSource =
    //     data.map((log) => ({
    //       key: log.id,
    //       remarks: log.remarks,
    //       creationTime: log.creationTime,
    //     })) || [];
    const dataSource = (data || []).map((log) => ({
        key: log.id,
        remarks: log.remarks,
        creationTime: log.creationTime,
    }));

    return (
        <Table
            dataSource={dataSource}
            loading={loading}
            columns={columns}
            pagination={false}
        />
    );
};

export default LogList;
