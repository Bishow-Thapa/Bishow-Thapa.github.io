    export function findUsersWithApproval(array, data) {
    if (!Array.isArray(data)) return [];

    // Step 1: Filter based on aspNetUserId and canApprove
    let foundUsers = data?.filter(
        (item) => array.includes(item?.aspNetUserId) && item?.canApprove === true
    );

    // Step 2: Map the filtered users to a new structure where 'id' is renamed to 'employeeId'
    let mappedUsers = foundUsers?.map((user) => ({
        employeeId: user.id, // Rename 'id' to 'employeeId'
        ...user, // Keep all other properties as they are
    }));

    return mappedUsers;
}

    1. Filter:

      The filter() method filters the data based on whether the aspNetUserId is found in the array and whether the canApprove property is true.

    2. Map:

      After filtering, we use map() to transform the structure of each object in the filtered array. Specifically, we rename id to employeeId and include the rest of the user properties using the spread operator (...user).

export function currentApprovers(data) {
  if (!Array.isArray(data)) return [];

  let approvers = [];
